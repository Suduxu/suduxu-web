# Configuration

Suduxu is configured through a `suduxu.json` file located next to the native runtime. Suduxu provides a service for creating `suduxu.json` files with custom configurations under [Suduxu Configurator](https://suduxu.com/configurator).

## Example Configuration

```json
{
  "server": {
    "address": "0.0.0.0",
    "port": 9000,
    "tcp_port": null,
    "udp_port": null,
    "file_port": null,
    "connection_strategy": "Open",
    "list": null,
    "rate_limit": {
      "enabled": false,
      "max_tcp_requests_per_minute": null
    }
  },
  "logging": {
    "debug_level": "Debug",
    "log_file": null,
    "max_log_size": null,
    "log_to_console": true
  },
  "security": {
    "enabled": false,
    "password": null
  },
  "file_sharing": {
    "enabled": false,
    "shared_directory": null,
    "files": null,
    "initially_loaded": null
  },
  "devices": {
    "initially_send_sensor_data": false,
    "max_devices": 4,
    "allowed_device_types": [
      "Android",
      "iOS",
      "Windows",
      "Linux",
      "macOS",
      "Other"
    ],
    "initial_frame_rate": 60
  },
  "screen_capture": {
    "enabled": false,
    "capture_on_server": false,
    "capture_directory": null
  },
  "sensors": {
    "accelerometer": true,
    "gyroscope": true,
    "magnetometer": true,
    "temperature": false,
    "humidity": false,
    "pressure": false,
    "light": false
  },
  "developer": {
    "prefer_cli": false,
    "allow_mocked_sensors": false,
    "allow_mocked_buttons": false
  },
  "health_check": {
    "server": {
      "enabled": false,
      "interval_ms": null,
      "timeout_ms": null
    },
    "client": {
      "enabled": false,
      "interval_ms": null,
      "timeout_ms": null
    }
  }
}
```

## `server`

Controls networking and client connections.

<TypeTable
  type={{
    address: {
      description: "IP address the server binds to.",
      type: "string",
      default: '"0.0.0.0"',
    },
    port: {
      description: "Shared port used for all services.",
      type: "integer",
      default: "9000",
    },
    tcp_port: {
      description: "Overrides the TCP port.",
      type: "integer | null",
      default: "null",
    },
    udp_port: {
      description: "Overrides the UDP port.",
      type: "integer | null",
      default: "null",
    },
    file_port: {
      description: "Overrides the file transfer port.",
      type: "integer | null",
      default: "null",
    },
    connection_strategy: {
      description:
        "Controls how the list field is interpreted. One of Open, Whitelist, Blacklist.",
      type: '"Open" | "Whitelist" | "Blacklist"',
      default: '"Open"',
    },
    list: {
      description:
        "List of allowed or blocked addresses depending on the selected connection_strategy.",
      type: "string[] | null",
      default: "null",
    },
  }}
/>

### `server.rate_limit`

TCP request rate limiting configuration.

<TypeTable
  type={{
    enabled: {
      description: "Enables rate limiting.",
      type: "boolean",
      default: "false",
    },
    max_tcp_requests_per_minute: {
      description: "Maximum allowed TCP requests per minute.",
      type: "integer | null",
      default: "null",
    },
  }}
/>

```json
{
  "rate_limit": {
    "enabled": true,
    "max_tcp_requests_per_minute": 1000
  }
}
```

## `health_check`

Connection health monitoring, configured separately for `server` and `client`.

### `health_check.server` / `health_check.client`

<TypeTable
  type={{
    enabled: {
      description: "Enable health checks.",
      type: "boolean",
      default: "false",
    },
    interval_ms: {
      description: "Heartbeat interval, in milliseconds.",
      type: "integer | null",
      default: "null",
    },
    timeout_ms: {
      description: "Timeout before disconnecting a peer, in milliseconds.",
      type: "integer | null",
      default: "null",
    },
  }}
/>

```json
{
  "client": {
    "enabled": true,
    "interval_ms": 1000,
    "timeout_ms": 5000
  },
  "server": {
    "enabled": true,
    "interval_ms": 1000,
    "timeout_ms": 5000
  }
}
```
